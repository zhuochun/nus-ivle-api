ivle

Login

ivle.login(key, redirectUrl); // return the login url
// result here

Get Token

ivle.getToken([url]); // return the token if it is found in location.href/url

User

Initial User

var user = ivle.User(key, token); // return a User instance

Validate User

user.validate(function(result) {
    // result = true/false
});

Get User Identities

Username

user.name(function(username) {
    // username = "WANG ZHUOCHUN"
});

Matric Number

user.id(function(id) {
    // id = "A0000000"
});

Email Address

user.email(function(email) {
    // email = "xxx@nus.edu.sg"
});

Unread Announcements

user.unreadAnnouncements(function(anns) {
    // anns = [...]
});

Search

var type = "Modules", q = {AcadYear: "2012/2013", Semester: 2};
user.search(type, q, function(result) {
    // result = [...]
});

Allowed Query Params:

// must have at least one query param in search
q = {
    ModuleCode: "ACC1002" // String
  , ModuleTitle: "Financial" // String
  , LecturerName: "XXX" // String
  , Department: "Arts" // String
  , Semester: "1" // String
  , AcadYear: "2012/2013" // String
  , ModNameExact: false // Boolean
  , LecNameExact: false // Boolean
};

Modules

Getting the Modules

var modules = [];
user.modules(function(mods) {
    modules = mods; // mods = Module[...]
});

Getting Data

var modId = modules[0].get("ID"); // ID of the 1st module in list

Update Module

modules[0].update(); // update the data of the 1st module

Announcements

modules[0].announcements(function(anns) {
    // anns = [...]
});

Workbins

modules[0].workbins(function(bins) {
    // bins = [...]
});

Forums

modules[0].forums(function(forums) {
    // forums = [...]
});

Webcasts

modules[0].webcasts(function(webcasts) {
    // webcasts = [...]
});

Gradebook

modules[0].gradebook(function(grades) {
    // grades = [...]
});
